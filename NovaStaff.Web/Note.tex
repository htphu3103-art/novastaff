src/
├── api/             # Chứa các file gọi API (axios instance, các hàm call API khách hàng,...)
├── components/      # Các thành phần giao diện nhỏ dùng chung (Button, Input, Table,...)
├── layouts/         # Chứa "khung" của trang web (giống Shared/_Layout.cshtml)
│   └── MainLayout/  # Layout chính gồm Sidebar, Header, Footer
├── pages/           # (Hoặc views) Chứa các trang hoàn chỉnh (Dashboard, Customer,...)
├── models/          # Định nghĩa các Interface/Type (Customer, PageItem,...)
├── routes/          # Cấu hình điều hướng (Router)
└── assets/          # Hình ảnh, icon, font
